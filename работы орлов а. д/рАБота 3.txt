#include <iostream>
using namespace std;
int main() {
	setlocale(0, "");

	cout << "[1] Плюс\n[2] Минус\n[3] Разелить\n[4] Умножить\n[5] Разделить со статоком\n";
	float x, y;
	int i, w;
	cin >> i;
	if (i == 1) {
		cout << "Первое число: ";
		cin >> x;
		cout << "Второе число: ";
		cin >> y;
		cout << "[+] Результат: " << x << " + " << y << " = " << x + y;
	}
	else if (i == 2) {
		cout << "Первое число: ";
		cin >> x;
		cout << "Второе число: ";
		cin >> y;
		cout << "[+] Результат: " << x << " - " << y << " = " << x - y;
	}
	else if (i == 3) {
		cout << "Первое число: ";
		cin >> x;
		cout << "Второе число: ";
		cin >> y;
		if (y == 0) {
			cout << "Делить на ноль нельзя.";
			return 0;
		}
		cout << "[+] Результат: " << x << " / " << y << " = " << x / y;
	}
	else if (i == 4) {
		cout << "Первое число: ";
		cin >> x;
		cout << "Второе число: ";
		cin >> y;
		cout << "[+] Результат: " << x << " * " << y << " = " << x * y;
	}
	else if (i == 5) {
		cout << "Первое число: ";
		cin >> i;
		cout << "Второе число: ";
		cin >> w;
		cout << "[+] Результат: " << i << " % " << w << " = " << i % w;
	}
	else {
		cout << "Такого знака нет.";
	}

	return 0;
}
